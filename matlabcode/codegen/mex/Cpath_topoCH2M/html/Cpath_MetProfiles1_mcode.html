1% Function Cpath_MetProfiles        Reads profile data along a path
 2% all the data is in hdf5 format
 3%
 4% NOTE: this version does not do interpolation
 5%     [topoData,ifWater,olat,olon,orange] = Cpath_MetProfiles(path_lat,path_lon,ranges);
 6%   input:
 7%         path_lat - vector of latitudes
 8%         path_lon - vector of longitudes
 9%         ranges - ranges from source
10%   outpu
11%         topoData - vector of topo data along the path
12%         ifWater - vector gives 1 for land, 0 for water
13%         olat,olon - vectors where topo data is obtained (because no interpolation)
14%         orange - output ranges (nearest profiles)
15
16function [topoData,ifWater,olat,olon,orange] = Cpath_MetProfiles(path_lat,path_lon,ranges);
17
18DatabasesDir = 'C:/Users/User/Downloads/Google Drive/Miscellaneous/IGPP/2015072118';
19filename = [DatabasesDir '/wrfout_d04_2015-07-21_12.GRM_F'];
20%filename = [DatabasesDir '/wrfout_d03_2015-07-21_12.GRM_F'];
21%filename = [DatabasesDir '/wrfout_d02_2015-07-21_12.GRM_F'];
22
23% explore what is in the file
24%dum = h5info(filename)
25% see the results of this in diary_hdf5
26%diary
27%h5disp(filename)
28%diary off
29
30deg2rad = pi/180;
31% convert to longitudes from -180 to 180 for this database
32path_lon=Csetminmax(path_lon,-180,180);         
33
34% read in grids of latitude, longitude, water (yes/no), terrain heights
35% 
36gridLat = h5read(filename,'/XLAT');             % latitude
37gridLat_units = h5readatt(filename,'/XLAT','units');
38gridLon = h5read(filename,'/XLONG');            % longitude
39gridLon_units = h5readatt(filename,'/XLONG','units');
40
41gridWater = h5read(filename,'/LANDMASK');   % LAND MASK (1 FOR LAND, 0 FOR WATER)
42gridElev = h5read(filename,'/HGT');         % Terrain height
43gridElev = h5read(filename,'/HGT');
44
45% -------------------------------------------------------------------------------
46% get topography data at nearest gridpoint for each point along the path
47% only if the data are being loaded at a new grid point
48% for each path point
49%   a) find the distance from the point to each gridpoint
50%   b) find the nearest gridpoint to the path point  
51%   c) if it differs from the last gridpoint, copy the profile 
52
53npath = length(path_lat); nprofs=0;
54latpt = 100; lonpt = 400;
55% -------------------------------------------------------------------------------
56% for each point along the path
57% -------------------------------------------------------------------------------
58for jj=1:npath
59% calculate the distance from the path point to all gridpoints in file
60    coslat = cos(path_lat(jj)*deg2rad);
61    xdist = coslat*(path_lon(jj)-gridLon);
62    ydist = (path_lat(jj)-gridLat);
63    dist = sqrt(xdist.*xdist+ydist.*ydist);
64% find the gridpoint closest to the current path point
65    dum = min(min(dist));
66    [irow,jcol] = find(dist==dum);
67    testlat = gridLat(irow,jcol); testlon = gridLon(irow,jcol);
68    
69% read the profile only if it is not at the same gridpoint as the last one
70    if (testlat ~= latpt | testlon ~= lonpt)
71        nprofs = nprofs+1;
72        latpt = testlat; lonpt = testlon;   % reset point
73        olat(nprofs) = testlat; olon(nprofs) = testlon;
74        topoData(nprofs) = gridElev(irow,jcol);
75        ifWater(nprofs) = gridWater(irow,jcol);
76        
77% now that I have a gridpoint, find the path point nearest to it
78% (I need this extra step to get an approximate range value)        
79        coslat = cos(testlat*deg2rad);
80        xdist = coslat*(path_lon-testlon);
81        ydist = (path_lat-testlat);
82        dist = sqrt(xdist.*xdist+ydist.*ydist); 
83        [dum,inr] = min(dist);  
84        orange(nprofs) = ranges(inr);
85
86%       [nprofs dum path_lat(jj) testlat path_lon(jj) testlon orange(nprofs)/1000]
87    end
88end 
89
90%plot the entrie grid of latitudes and longitudes,
91% with picked stations along the path
92figure(11),clf,plot(gridLon,gridLat,'k.')
93hold on,plot(olon,olat,'ro')
94plot(path_lon(1),path_lat(1),'gp','MarkerSize',12,'MarkerFaceColor','g')
95%keyboard
