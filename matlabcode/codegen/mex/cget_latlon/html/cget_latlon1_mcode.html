1%CGET_LATLON        Get latitudes and longitudes along a line
  2%
  3%       function [plat,plon]=cget_latlon(blat,blon,ranges,azm,major_axis);
  4%
  5% This is a translation to matlab of the USGS PROJ-4.4 geographic
  6% projection library to compute positions along a series of ranges
  7% along a given azimuth from a reference point.
  8% Uses ellipsoidal earth model set to Clark 1966 standard.
  9%
 10% Inputs:
 11%       blat,blon       lat,lon coordinates of start point
 12%                       in degrees (scalar)
 13%
 14%       ranges          range in nmi (scalar or vector)
 15%       azm             single-valued (same size as range) (degrees)
 16%       major_axes - skip for an elliptical earth, set=0 for spherical Earth
 17%
 18% Outputs:
 19%       plat,plon       lat,lons along path
 20%
 21
 22
 23function [plat,plon]=cget_latlon(blat,blon,ranges,azm,major_axis,esquared);
 24
 25coder.extrinsic('keyboard');
 26signS = 0;
 27merid = 0;
 28cosa12 = 0;
 29M = 0;
 30c1 = 0;
 31c2 = 0;
 32D = 0;
 33s1 = 0;
 34P = 0;
 35ds = 0;
 36phi2 = 0;
 37de = 0;
 38ss = 0;
 39
 40
 41if nargin < 5
 42% Set the parameters for the "Clark 1966" ellipsoid model
 43   major_axis =  6378206.4;        % earth major axis in meters
 44   esquared   = 0.006768658;       % eccentricity squared
 45end
 46
 47if major_axis == 0
 48   major_axis =  6378206.4;
 49   esquared   = 0;
 50end
 51
 52keyboard
 53
 54%~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
 55% now do the stuff that used to be in .c
 56
 57len=length(ranges);
 58if length(azm)~=len
 59   disp('RANGES and AZM must be the same length')
 60   return
 61end
 62
 63deg2rad=pi/180;
 64rad2deg=180/pi;
 65halfpi=pi/2;
 66dtol=1e-12;
 67meritol=1e-10;
 68
 69a=major_axis;S=ranges*1852;
 70phi1=blat*deg2rad;lam1=blon*deg2rad;al12=azm*deg2rad;
 71es=esquared;
 72
 73if es ~= 0.0
 74   ellipse = 1;
 75else
 76   ellipse = 0;
 77end
 78
 79if ellipse == 1
 80   onef= sqrt(1. - es);
 81   f  = 1. - onef;
 82   f4 = f/4.;
 83else
 84   onef = 1.;f=0;f4=0;
 85end
 86
 87al12=cadjlon(al12);
 88is1=find(abs(al12) > halfpi);signS(is1)=1;
 89is0=find(abs(al12) <= halfpi);signS(is0)=0;
 90if ellipse == 1
 91   th1=atan(onef * tan(phi1));
 92else
 93   th1=phi1;
 94end
 95costh1 = cos(th1);sinth1 = sin(th1);
 96sina12 = sin(al12);
 97im1=find(abs(sina12) < meritol);merid(im1)=1;
 98im0=find(abs(sina12) >= meritol);merid(im0)=0;
 99for j=1:len
100   if merid(j) == 1
101      sina12(j) = 0.;
102      if abs(al12(j)) < halfpi
103         cosa12(j)=1;
104      else
105         cosa12(j)=-1;
106      end
107      M(j) = 0.;
108   else 
109      cosa12(j)= cos(al12(j));
110      M(j)= costh1 * sina12(j);
111   end
112end
113N = costh1 * cosa12;
114if ellipse ==1
115%   if merid(j) == 1
116      c1(im1)= 0.; c2(im1)= f4; D(im1)= 1.-c2(im1); 
117      D(im1) = D(im1).*D(im1); P(im1) = c2(im1)./D(im1);
118%   else 
119      c1(im0) = f*M(im0);c2(im0)= f4*(1. - M(im0).*M(im0));
120      D(im0) = (1.-c2(im0)).*(1.-c2(im0)-c1(im0).*M(im0));
121      P(im0) = (1.+.5*c1(im0).*M(im0)).*c2(im0)./D(im0);
122%   end
123end
124for j=1:len
125   if merid(j) == 1
126      s1(j) = halfpi - th1;
127   else 
128      if abs(M(j)) >= 1
129         s1(j) = 0;
130      else
131         s1(j) = acos(M(j));
132      end
133      s1(j)=  sinth1/sin(s1(j));
134      if abs(s1(j)) >=1
135         s1(j) = 0;
136      else
137         s1(j) = acos(s1(j));
138      end
139   end
140end
141
142if ellipse == 1
143   d = S./(D * a); 
144   d(is1) = -d(is1);
145   u = 2.*(s1 - d);
146   V = cos(u + d);
147   sind = sin(d);
148   X = c2.*c2.*sind.*cos(d).*(2. * V.*V - 1.);
149   ds = d + X - 2. * P.*V.*(1. - 2. * P.*cos(u)).*sind;
150   ss = s1 + s1 - ds;
151else 
152   ds = S/a;
153   ds(is1) = - ds(is1);
154end
155cosds = cos(ds); sinds = sin(ds);
156sinds(is1) = - sinds(is1);
157al21 = N.*cosds - sinth1 * sinds;
158for j=1:len
159   if merid(j)==1
160      phi2(j) = atan(tan(halfpi + s1(j)- ds(j))/onef);
161      if al21(j) > 0.
162         al21(j) = pi;
163         if signS(j)==1; 
164            de(j) = pi;
165         else 
166            phi2(j) = - phi2(j); de(j) = 0.;
167         end              
168      else 
169         al21(j) = 0.;
170         if signS(j)==1;
171            phi2(j) = - phi2(j);de(j) = 0;
172         else
173            de(j) = pi;    
174         end
175      end          
176   else 
177      al21(j) = atan(M(j)/al21(j));
178      if al21(j) > 0; al21(j) = al21(j) + pi; end
179      if al12(j) < 0.; al21(j) = al21(j) - pi; end
180      al21(j) = cadjlon(al21(j));
181      if ellipse == 1
182         denom = onef * M(j);
183      else
184         denom = M(j);
185      end
186      phi2(j)=atan(-(sinth1*cosds(j)+N(j)*sinds(j))*sin(al21(j))/denom);
187      de(j)=atan2(sinds(j)*sina12(j),...
188                 (costh1*cosds(j)-sinth1*sinds(j)*cosa12(j)));
189      if ellipse == 1
190         if signS(j)==1
191            de(j)=de(j)+c1(j)*((1.-c2(j))*ds(j)+c2(j)*sinds(j)*cos(ss(j)));
192         else
193            de(j)=de(j)-c1(j)*((1.-c2(j))*ds(j)-c2(j)*sinds(j)*cos(ss(j)));
194         end
195      end
196   end
197end
198lam2 = cadjlon(lam1+de);
199
200lam1=lam1*rad2deg;
201phi1 = phi1*rad2deg;
202al12 = al12*rad2deg;
203plon = lam2*rad2deg;
204plat = phi2*rad2deg;
205
