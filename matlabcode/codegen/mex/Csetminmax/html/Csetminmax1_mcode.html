1% Function CSETMINMAX   resets value to within given limits
 2%   useful for ensuring that longitude is in correct range
 3%      [value] = Csetminmax(value,lowlim,uplim)
 4%   
 5%   input:
 6%       value - value to be limited
 7%       lolim = lower limit
 8%       uplimit = upper limit
 9%   output:
10%       value - new value (between limits)
11%
12%   EXAMPLE:
13%   a) force longitude 0<=longitude<360
14%       longitude = -180;
15%       [longitude] = Csetminmax(longitude,0,360)
16%   b) force longitude -pi<=longitude<pi
17%       longitude = 2*pi;
18%       [longitude] = Csetminmax(longitude,-pi,pi)
19
20function    [value] = Csetminmax(value,lowlim,uplim);
21
22diff=uplim-lowlim;
23if diff <=0
24    disp('ERROR: upper limit must be higher than lower limit')
25    return
26end
27
28ind = find(value>=uplim);
29while isempty(ind)==0
30    value(ind) = value(ind)-diff;
31    ind = find(value>=uplim);
32end
33
34ind = find(value<lowlim);
35while isempty(ind)==0
36    value(ind) = value(ind)+diff;
37    ind = find(value<lowlim);
38end
39
40return
