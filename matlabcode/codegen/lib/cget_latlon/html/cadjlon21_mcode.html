1%CADJLON        reduces argument to range from -pi to pi for single value, 
 2% use Csetminmax instead 
 3%
 4%       function [theta]=cadjlon(theta);
 5%
 6
 7function [theta]=cadjlon(theta);
 8
 9twopi=2*pi;
10
11ind=find(theta>pi);
12while isempty(ind)==0
13      theta(ind) = theta(ind)-twopi;
14      ind=find(theta>pi);
15end
16
17ind=find(theta< -pi);
18while isempty(ind)==0
19      theta(ind) = theta(ind)+twopi;
20      ind=find(theta< -pi);
21end
